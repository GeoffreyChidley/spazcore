 12 /**
 13  * Given a string, this returns an XMLDocument
 14  * @param {string} string an xml string
 15  * @return {XMLDocument}
 16  */
 17 sc.helpers.createXMLFromString = function (string) {
 18 	var xmlParser, xmlDocument;
 19 	try {
 20 		xmlParser = new DOMParser();
 21 		xmlDocument = xmlParser.parseFromString(string, 'text/xml');
 22 		return xmlDocument;
 23 	} catch (e) {
 24 		sc.helpers.dump(e.name + ":" + e.message);
 25 		return null;
 26 	}
 27 };
