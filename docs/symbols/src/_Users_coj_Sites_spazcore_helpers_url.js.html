1 /**
  2  * Takes a key/val pair object and returns a query string 
  3  * @member sc.helpers 
  4  */
  5 sc.helpers.objectToQueryString = function(object) {
  6 	var query_string, key, val, pieces = [];
  7 	
  8 	for(key in object) {
  9 		val = object[key];
 10 		pieces.push(encodeURIComponent(key)+'='+encodeURIComponent(val));
 11 	}
 12 	query_string = pieces.join('&');
 13 	return query_string;
 14 };
